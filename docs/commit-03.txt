// Modify the addTask function:
function addTask() {
    const taskInput = document.getElementById('taskInput');
    const prioritySelect = document.getElementById('prioritySelect');
    const taskText = taskInput.value.trim();

    if (taskText === '') {
        alert('Please enter a task!');
        return;
    }

    const newTask = {
        id: Date.now(),
        text: taskText,
        priority: prioritySelect.value,  // Add this line
        completed: false,
        createdAt: new Date().toLocaleDateString()  // Add this line
    };

    tasks.push(newTask);
    renderTasks();
    taskInput.value = '';
}
